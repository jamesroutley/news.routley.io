// It's a classic for a reason
fn fib(n: i32) -> i32 {
    if n <= 1 {
        n
    } else {
        fib(n - 1) + fib(n - 2)
    }
}

fn main() -> i32 {
    // Print the first 20 Fibonacci numbers
    let mut i = 0;
    while i < 20 {
        @dbg(fib(i));
        i = i + 1;
    }

    // Return fib(10) = 55
    fib(10)
}
