fun rect_like_to_rect(v):
  match v
  | Rect(_, _, _, _): v
  | {"LT": [l, t], "RB": [r, b]}: Rect(l, t, r, b)
  | {"TL": [t, l], "RB": [b, r]}: Rect(l, t, r, b)

class Rect(left, top, right, bottom)
 
bind.macro 'OriginRect($right, $bottom)':
  'Rect(0, 0, $right, $bottom)'
 
fun area(r):
  match r
  | OriginRect(r, b): r*b
  | Rect(l, t, r, b): (r-l)*(b-t)
