// Reliably check out a customer order
// DBOS guarantees each step executes once and only once

@Workflow()
static async checkoutOrder(ctxt: WorkflowContext, order: Order) {
    // Reserve inventory for the order
    await ctxt.invoke(Shop).reserveInventory(order); 
    // Submit payment to a third-party processor
    await ctxt.invoke(Shop).submitPayment(order); 
    // Await a callback with payment status
    const paymentStatus = await ctxt.recv<string>(paymentCallback)
    // Fulfill paid orders, cancel unpaid orders and return inventory
    if (paymentStatus === Status.PAID) {
      await ctxt.invoke(Shop).fulfillOrder(order);
    } else {
      await ctxt.invoke(Shop).cancelOrder(order);
      await ctxt.invoke(Shop).undoReserveInventory(order);
    }
  }

  • OS services implemented on top of a distributed DBMS
  • Built-in transactional, fault-tolerant state management
  • Simplifies the stack: no need for containers, cluster management, workflow orchestration,...
  • Seamless scaling, high performance, and high availability
  • Metrics, logs, traces stored in SQL-accessible tables
  • Smaller cyber attack surface, cyberattack self-detection, and cyber-resilience
