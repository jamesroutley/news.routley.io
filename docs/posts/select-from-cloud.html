select
  aws.name aws_user_name,
  slack.id as slack_user_id,
  slack.display_name as slack_name
from
  aws_iam_user as aws,
  slack_user as slack
where
  aws.name = slack.email;

  group_name,
  group_id

  and cidr_ip = '0.0.0.0/0';

  id,
  name
from
  azure_compute_image

  tags -> 'owner' is null or

  volume_id,
  volume_type
from 
  aws_ebs_volume
where
  attachments is null;
